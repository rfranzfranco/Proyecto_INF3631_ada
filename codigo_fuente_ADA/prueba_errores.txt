use Ada.Text_IO;
use Ada.Integer_Text_IO;
use Ada.Float_Text_IO;

procedure Prueba_Errores is
   -- Variables bien declaradas
   Valor : Integer := 100;
   Factor : Float := 2.5;
   Activo : Boolean := True;
   
   -- ERROR SEMANTICO 1: Variable duplicada en el mismo ambito
   Valor : Float := 5.5;
   
   -- ERROR LEXICO: Caracter no permitido '@' y decimal mal formado
   Extraño : Character := '@';
   DecimalMal : Float := 10.a;
   
   -- ERROR SEMANTICO 2: La constante no esta inicializada
   Limite : constant Integer;

begin
   Put_Line ("Iniciando prueba con errores...");
   
   -- ERROR SEMANTICO 3: Incompatibilidad de tipos en asignacion (Integer := Float)
   Valor := Factor;
   
   -- ERROR SEMANTICO 4: Asignacion a una constante
   Limite := 50;

   -- ERROR SINTACTICO: Falta punto y coma
   Valor := Valor + 5

   -- ERROR SEMANTICO 5: Variable no declarada (no existe)
   NoExiste := 42;

end Prueba_Errores;
